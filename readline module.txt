const readline = require("readline");

const rl = readline.createInterface({
	input: process.stdin,
    output: process.stdout,
});
    
let input = [];
    
rl.on("line", (line)=>{
    input = line.split(' ').map(element => parseInt(element));
   	rl.close();
});
    
rl.on("close", ()=>{
/*입력이 끝나고 실행할 코드*/
	process.exit();
});
